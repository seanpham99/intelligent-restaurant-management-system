# Backend-Frontend Contract Integration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Rewrite backend contracts to match frontend needs, then integrate frontend screens with real backend menu/order/status APIs.

**Architecture:** Keep gateway as the public API boundary and make shared backend DTOs the canonical contract source. Extend menu/order/status contracts at backend first, then build a typed frontend API layer and wire existing screen-state flow (`Welcome -> Menu -> Confirmation -> Success -> Payment`) to real responses/events.

**Tech Stack:** FastAPI, SQLAlchemy, Docker Compose, MQTT/WebSocket bridge, React 19, TypeScript, Vite.

```ts
// frontend/src/api/client.ts
const API_BASE_URL = import.meta.env.VITE_API_BASE_URL ?? 'http://localhost:8000';

export async function getJson<T>(path: string): Promise<T> {
  const res = await fetch(`${API_BASE_URL}${path}`);
  if (!res.ok) throw new Error(`GET ${path} failed: ${res.status}`);
  return res.json() as Promise<T>;
}

export async function postJson<TReq, TRes>(path: string, body: TReq): Promise<TRes> {
  const res = await fetch(`${API_BASE_URL}${path}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(body),
  });
  if (!res.ok) throw new Error(`POST ${path} failed: ${res.status}`);
  return res.json() as Promise<TRes>;
}
```

```ts
import type { OrderCreateInput, OrderCreateResponse, OrderStatusEvent } from '../types';
import { postJson } from './client';

export const createOrder = (body: OrderCreateInput[]) =>
  postJson<OrderCreateInput[], OrderCreateResponse[]>('/order/create', body);

export function openOrderStatusSocket(onMessage: (event: OrderStatusEvent) => void): WebSocket {
  const ws = new WebSocket((import.meta.env.VITE_WS_BASE_URL ?? 'ws://localhost:8000') + '/order/status');
  ws.onmessage = (e) => onMessage(JSON.parse(e.data) as OrderStatusEvent);
  return ws;
}
```
